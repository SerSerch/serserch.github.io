## lesson 3
``` bush
// список файлов
ls
// список  файлов вместе с скрытыми
ls -a
// создать директорию
mkdir "[name]"
// переход в директорию
cd /D "[path]"
// создать файл
touch "[name]"
// копировать файл
cp "[name]" "[name_copy]"
// переименовать файл
mv "[name]" "[new_name]"
// вывод текста в файл
echo "[text]" > "[name]"
// чтение файла
cat "[name]"
// полное удаление файла
rm "[name]"
// полное удаление директории
rm -R "[name]"
```
